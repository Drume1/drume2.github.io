是那个需要我们修复的rsa密钥文件

写个脚本帮助我们去修复一下

import subprocess
import itertools
import string
import os

# 配置区（使用前必须修改！）
IP = "目标服务器IP"         # 替换为实际IP
USERNAME = "用户名"         # SSH用户名
DAMAGED_KEY_FILE = "id_rsa" # 损坏的私钥文件路径
TIMEOUT = 3                # SSH连接超时（秒）

def validate_key(replacement):
    """验证替换后的私钥是否有效"""
    try:
        # 1. 读取原始私钥内容
        with open(DAMAGED_KEY_FILE) as f:
            key_content = f.read().replace("***", replacement)
        
        # 2. 创建临时私钥文件
        temp_key = f"temp_key_{replacement}"
        with open(temp_key, "w") as f:
            f.write(key_content)
        os.chmod(temp_key, 0o600)  # 关键：设置文件权限
        
        # 3. 快速格式验证
        keygen_test = subprocess.run(
            ["ssh-keygen", "-y", "-f", temp_key],
            stderr=subprocess.PIPE,
            stdout=subprocess.PIPE,
            timeout=2
        )
        if keygen_test.returncode != 0:
            return False
            
        # 4. 实际SSH连接测试
        ssh_test = subprocess.run(
            [
                "ssh", "-i", temp_key,
                "-o", "StrictHostKeyChecking=no",
                "-o", "ConnectTimeout={}".format(TIMEOUT),
                "{}@{}".format(USERNAME, IP),
                "echo 'SSH连接成功！'"
            ],
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            timeout=TIMEOUT
        )
        return ssh_test.returncode == 0
        
    finally:
        # 清理临时文件
        if os.path.exists(temp_key):
            os.unlink(temp_key)

def repair_private_key():
    """修复掩码的私钥文件"""
    # 定义字符集（大写字母+数字）
    char_pool = string.ascii_uppercase + string.digits
    total_combinations = len(char_pool) ** 3
    
    print(f"开始暴力破解，共 {total_combinations} 种组合...")
    print("可能需要较长时间，请耐心等待...")
    
    # 遍历所有3字符组合
    for i, combo in enumerate(itertools.product(char_pool, repeat=3)):
        test_str = ''.join(combo)
        
        # 显示进度（每100种组合）
        if i % 100 == 0:
            progress = i / total_combinations * 100
            print(f"\r当前进度: {i}/{total_combinations} ({progress:.1f}%)", end="", flush=True)
        
        # 验证当前组合
        if validate_key(test_str):
            print(f"\n\n✅ 发现有效组合: {test_str}")
            
            # 生成修复后的私钥文件
            with open(DAMAGED_KEY_FILE) as f:
                fixed_content = f.read().replace("***", test_str)
            
            fixed_filename = f"{DAMAGED_KEY_FILE}.repaired"
            with open(fixed_filename, "w") as f:
                f.write(fixed_content)
            os.chmod(fixed_filename, 0o600)
            
            print(f"修复后的私钥已保存至: {fixed_filename}")
            print(f"请使用命令测试: ssh -i {fixed_filename} {USERNAME}@{IP}")
            return
    
    print("\n\n❌ 未找到有效组合，请检查：")
    print("1. IP地址和用户名是否正确")
    print("2. 原始私钥文件路径是否正确")
    print("3. 服务器SSH服务状态")
    print("4. 尝试扩展字符集（添加小写字母或符号）")

if __name__ == "__main__":
    # 运行前提醒
    print("⚠️ 使用前请务必修改配置区的IP/用户名/私钥路径！")
    print("-" * 50)
    repair_private_key()
