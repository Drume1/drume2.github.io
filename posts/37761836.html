import hashlib
def calc_pass(username, target_hash):
    # 测试常见密码
    common_passwords = [
        '123456789', 'password', 'qwerty', 'admin123', 'welcome',
        'sharpnet', 'maxwell', 'morrison', 'adminadmin', 'employee',
        'system', 'record', 'wave', 'tech', 'sharpadmin'
    ]
    for pwd in common_passwords:
        # 跳过小于6位的密码
        if len(pwd) < 6:
            continue
        # 计算哈希：md5(md5(pass).md5(user))
        step1 = hashlib.md5(pwd.encode()).hexdigest()
        step2 = hashlib.md5(username.encode()).hexdigest()
        total = step1 + step2
        final_hash = hashlib.md5(total.encode()).hexdigest()
        if final_hash == target_hash:
            return pwd
    return None
pass_xxx2xy = calc_pass("xxx2xy", "10a55271c201e41913764ff95b33248b")
print(f"Password for xxx2xy: {pass_xxx2xy}")

pass_admins = calc_pass("admins", "02adcdf2171dc7e5757cdd7c0b91fa03")
print(f"Password for admins: {pass_admins}")
